# Phase 5.4: Post-Training Inference Validation

This guide explains how to perform inference using your newly trained LoRA models and compare the results with the pre-training baseline.

---

## 1. How LoRA Loading Works in DiffSynth-Studio
DiffSynth-Studio's pipeline inherits from `BasePipeline` which includes a standard `load_lora` method to patch the base models. 

To load and use your trained LoRA:
```python
# For Z-Image (DiT architecture):
pipe.load_lora(
    module=pipe.dit, 
    lora_config="path/to/Z-Image_lora/epoch_5.safetensors"
)

# For Flux (DiT architecture):
pipe.load_lora(
    module=pipe.dit, 
    lora_config="path/to/FLUX.1-dev_lora/epoch_5.safetensors"
)

# For SDXL (UNet architecture):
pipe.load_lora(
    module=pipe.unet, 
    lora_config="path/to/stable-diffusion-xl-base-1.0_lora/epoch_5.safetensors"
)
```

## 2. Complete Validation Script (`test_lora_inference.py`)

Create a python script named `test_lora_inference.py` in your `/home/project_id_10/Captioning/phase3/` directory to run inference on the trained LoRAs:

```python
#!/usr/bin/env python3
import os
import sys
import json
import argparse
import torch
from pathlib import Path

# Add DiffSynth-Studio to path
PROJECT_ROOT = Path(__file__).resolve().parents[1]
sys.path.append(str(PROJECT_ROOT / "DiffSynth-Studio"))

from diffsynth.pipelines.z_image import ZImagePipeline, ModelConfig as ZModelConfig
from diffsynth.pipelines.flux_image import FluxImagePipeline, ModelConfig as FluxModelConfig
from diffsynth.pipelines.sdxl_image import SDXLImagePipeline, ModelConfig as SDXLModelConfig

def load_record(jsonl_path, index):
    with open(jsonl_path, "r", encoding="utf-8") as f:
        lines = f.readlines()
    if index >= len(lines):
        print(f"[ERROR] Index {index} out of bounds (file has {len(lines)} records).")
        sys.exit(1)
    record = json.loads(lines[index].strip())
    # Handle either 'prompt' or 'caption' keys robustly
    record["prompt"] = record.get("prompt") or record.get("caption")
    return record

# ===== Z-Image LoRA Inference =====
def run_zimage_lora(prompt: str, lora_path: str, output_path: str):
    print(f"\n--- Running Z-Image LoRA Inference ---")
    pipe = ZImagePipeline.from_pretrained(
        torch_dtype=torch.bfloat16,
        device="cuda",
        model_configs=[
            ZModelConfig(model_id="Tongyi-MAI/Z-Image", origin_file_pattern="transformer/*.safetensors"),
            ZModelConfig(model_id="Tongyi-MAI/Z-Image-Turbo", origin_file_pattern="text_encoder/*.safetensors"),
            ZModelConfig(model_id="Tongyi-MAI/Z-Image-Turbo", origin_file_pattern="vae/diffusion_pytorch_model.safetensors"),
        ],
        tokenizer_config=ZModelConfig(model_id="Tongyi-MAI/Z-Image-Turbo", origin_file_pattern="tokenizer/"),
    )
    
    print(f"Loading LoRA weights from: {lora_path}")
    pipe.load_lora(module=pipe.dit, lora_config=lora_path)
    
    image = pipe(
        prompt=prompt,
        seed=42,
        rand_device="cuda",
        num_inference_steps=50,
        cfg_scale=4.0,
        width=1024,
        height=1024
    )
    image.save(output_path)
    print(f"[OK] LoRA Image saved to: {output_path}")

# ===== FLUX.1-dev LoRA Inference =====
def run_flux_lora(prompt: str, lora_path: str, output_path: str):
    print(f"\n--- Running FLUX.1-dev LoRA Inference ---")
    pipe = FluxImagePipeline.from_pretrained(
        torch_dtype=torch.bfloat16,
        device="cuda",
        model_configs=[
            FluxModelConfig(model_id="black-forest-labs/FLUX.1-dev", origin_file_pattern="flux1-dev.safetensors"),
            FluxModelConfig(model_id="black-forest-labs/FLUX.1-dev", origin_file_pattern="text_encoder/model.safetensors"),
            FluxModelConfig(model_id="black-forest-labs/FLUX.1-dev", origin_file_pattern="text_encoder_2/*.safetensors"),
            FluxModelConfig(model_id="black-forest-labs/FLUX.1-dev", origin_file_pattern="ae.safetensors")
        ]
    )
    
    print(f"Loading LoRA weights from: {lora_path}")
    pipe.load_lora(module=pipe.dit, lora_config=lora_path)
    
    image = pipe(
        prompt=prompt,
        seed=42,
        rand_device="cuda",
        num_inference_steps=20,
        cfg_scale=3.5,
        width=1024,
        height=1024
    )
    image.save(output_path)
    print(f"[OK] LoRA Image saved to: {output_path}")

# ===== SDXL LoRA Inference =====
def run_sdxl_lora(prompt: str, lora_path: str, output_path: str):
    print(f"\n--- Running SDXL LoRA Inference ---")
    pipe = SDXLImagePipeline.from_pretrained(
        torch_dtype=torch.float16,
        device="cuda",
        model_configs=[
            SDXLModelConfig(model_id="stabilityai/stable-diffusion-xl-base-1.0", origin_file_pattern="unet/diffusion_pytorch_model.safetensors"),
            SDXLModelConfig(model_id="stabilityai/stable-diffusion-xl-base-1.0", origin_file_pattern="text_encoder/model.safetensors"),
            SDXLModelConfig(model_id="stabilityai/stable-diffusion-xl-base-1.0", origin_file_pattern="text_encoder_2/model.safetensors"),
            SDXLModelConfig(model_id="stabilityai/stable-diffusion-xl-base-1.0", origin_file_pattern="vae/diffusion_pytorch_model.safetensors"),
        ]
    )
    
    print(f"Loading LoRA weights from: {lora_path}")
    pipe.load_lora(module=pipe.unet, lora_config=lora_path)
    
    image = pipe(
        prompt=prompt,
        seed=42,
        rand_device="cuda",
        num_inference_steps=50,
        cfg_scale=7.5,
        width=1024,
        height=1024
    )
    image.save(output_path)
    print(f"[OK] LoRA Image saved to: {output_path}")

def main():
    parser = argparse.ArgumentParser(description="DiffSynth LoRA Inference Validator")
    parser.add_argument("--model", type=str, choices=["zimage", "flux", "sdxl"], required=True)
    parser.add_argument("--index", type=int, default=0)
    parser.add_argument("--epoch", type=int, default=5, help="Epoch checkpoint to validate (default: 5)")
    args = parser.parse_args()

    jsonl_path = PROJECT_ROOT / "data" / "id10" / args.model / f"metadata_{args.model}.jsonl"
    record = load_record(jsonl_path, args.index)
    prompt = record.get("prompt")
    task_id = record.get("metadata", {}).get("task_id", "unknown")
    
    lora_path = str(PROJECT_ROOT / "data" / "id10" / args.model / "models" / f"{args.model}_lora" / f"epoch_{args.epoch}.safetensors")
    
    # Custom naming for SDXL lora file mapping
    if args.model == "sdxl":
        lora_path = str(PROJECT_ROOT / "data" / "id10" / "sdxl" / "models" / "stable-diffusion-xl-base-1.0_lora" / f"epoch_{args.epoch}.safetensors")
    elif args.model == "zimage":
        lora_path = str(PROJECT_ROOT / "data" / "id10" / "zimage" / "models" / "Z-Image_lora" / f"epoch_{args.epoch}.safetensors")
    elif args.model == "flux":
        lora_path = str(PROJECT_ROOT / "data" / "id10" / "flux" / "models" / "FLUX.1-dev_lora" / f"epoch_{args.epoch}.safetensors")

    output_dir = PROJECT_ROOT / "data" / "lora_outputs"
    output_dir.mkdir(parents=True, exist_ok=True)
    output_path = str(output_dir / f"lora_{args.model}_task_{task_id}_epoch_{args.epoch}.png")

    print(f"Loaded Task ID: {task_id}")
    print(f"Prompt: {prompt}")

    if args.model == "flux":
        run_flux_lora(prompt, lora_path, output_path)
    elif args.model == "sdxl":
        run_sdxl_lora(prompt, lora_path, output_path)
    elif args.model == "zimage":
        run_zimage_lora(prompt, lora_path, output_path)

if __name__ == "__main__":
    main()
```

## 3. Run Inference Validation
After training is finished and you have generated the LoRA checkpoints (e.g. `epoch_5.safetensors`), run the validation script for the desired model:

```bash
# Validate Z-Image LoRA output for Task ID 1 (index 0)
python phase3/test_lora_inference.py --model zimage --index 0 --epoch 5

# Validate Flux LoRA output
python phase3/test_lora_inference.py --model flux --index 0 --epoch 5

# Validate SDXL LoRA output
python phase3/test_lora_inference.py --model sdxl --index 0 --epoch 5
```

Compare the resulting image files in `data/lora_outputs/` directly with the corresponding baseline image in `data/baseline_outputs/` to visually evaluate style learning!
